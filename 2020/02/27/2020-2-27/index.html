模板例子

Vue.component('button-counter', {
            data: function () {
                return {
                    count: 0
                }
            },
            template: '<button @click="handle">点击了{{count}}次</button>',
            methods: {
                handle: function () {
                    this.count+=2;
                }
            }
        })

注意

1.data 必须是一个函数

2.组件模板内容必须是单个根元素

3.可以使用模板字符串

4.最好使用短横线的命名方式

5.局部组件只能在注册他的父组件下使用

局部注册方式

var HelloWorld = {
           data: function () {
               return {
                   msg: "hello,I'm message"
               }
           },
           template: `<div>msg</div>`,
           methods: {

           }
       }
       var vm = new Vue({
           el: '#app',
           data: {},
           methods: {},
           components: {
               'hello-world': HelloWorld

           }
       })

父组件向子组件传递值

1,组件内部通过props接受传递过来的值

<body>
    <div id="app">
        <div>{{pmsg}}</div>
        <menu-item :title='ptitle' content='hello'></menu-item>
    </div>

    <script>
        Vue.component('menu-item', {
            props:['title','content'],
            data: function () {
                return {
                    msg: '子组件本身的数据'
                }
            },
            template: `<div>{{msg+'----'+title+'---'+content}}</div>`

        })
        var vm = new Vue({
            el: '#app',
            data: {
                pmsg: '父组件的内容',
                ptitle:'动态属性绑定'
            },
            methods: {}
        });
    </script>
</body>

2.父组件传递数据给子组件

<body>
    <div id="app">
        <menu-list :arr='arr' :obj='pobj' :num='num'>
            <menu-list />
    </div>

    <script>
        Vue.component('menu-list', {
            props: ['arr', 'obj', 'num'],
            data: function () {
                return {
                    msg: 'hello'
                }
            },
            template: `<div><div v-for='item in arr'>{{item}}</div></div>`
        })
        var vm = new Vue({
            el: '#app',
            data: {
                arr: [1, 2, 3, 4],
                pobj: {
                    id: 'angelina',
                    name: 12,

                },
                num: 'hello'
            },
            methods: {}
        });
    </script>
</body>

</html>

3.子组件传递数据给父组件

<body>
    <div id="app">
        <div :style='{fontSize:fontSize+"px"}'>{{pmsg}}</div>
        <my-com1 @enlarge-text='handle($event)'>
            </mycon1>
    </div>

    <script>
        Vue.component('my-com1', {
            props: [],
            template: `
            <div>
                <button @click='$emit("enlarge-text",5)'>click me</button>
                <button @click='$emit("enlarge-text",20)'>click me</button>
            </div>
            `

        })
        var vm = new Vue({
            el: '#app',
            data: {
                pmsg: '父组件中内容',
                fontSize: 10
            },
            methods: {
                handle(val) {
                    this.fontSize += val;
                }
            }
        });
    </script>
</body>
