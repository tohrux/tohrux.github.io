👩computed和methods的区别:

conputed里的方法只会执行一次,然后把结果缓存起来,下次直接使用这个结果

methods每调用一次就执行一次,method在插值表达式中调用时要加上()

侦听器

应用场景:

数据变化时执行异步或开销较大的操作

触发方式:

数据一变化就通知侦听器绑定方法

var vm = new Vue({
            el: '#app',
            data: {
                firstname:'anglina',
                lastname:'Me',
                fullname: 'angelina Me',
            },
            methods: {},
            watch: {
                firstname:function(val){
                    this.fullname = val + ' ' + this.lastname;
                },
                lastname:function(val){
                    this.fullname = this.firstname + ' ' + val;
                }
            },
        });

computed的实现方式

var vm = new Vue({
            el: '#app',
            data: {
                firstname: 'anglina',
                lastname: 'Me',
                // fullname: 'angelina Me',
            },
            methods: {},
            computed: {
                fullname: function () {
                    return this.firstname + this.lastname;
                }
            }

侦听器的应用场景

<body>
    <div id="app">
        <input type="text" v-model='uname'>
        <div>
            {{msg}}
        </div>

    </div>

    <script>
        var vm = new Vue({
            el: '#app',
            data: {
                uname: '',
                msg: '请输入你的名字'
            },
            methods: {
                check(message) {
                    var that = this;
                    setTimeout(function () {
                        if (message === 'angelina') {
                            that.msg = '已经被占用了';
                        } else {
                            that.msg = 'congratulation';
                        }
                    }, 1000)
                }
            },
            watch: {
                uname: function (val) {
                    this.check(val);
                    this.msg = 'loading';

                }
            },
        });
    </script>
</body>

过滤器

<body>
    <div id="app">
        {{msg|msgFormat}}
    </div>

    <script>
        var vm = new Vue({
            el: '#app',
            data: {
                msg: 'hello'
            },
            methods: {},
            filters: {
                msgFormat: function (val) {
                    return val.charAt(0).toUpperCase() + val.slice(1);
                }
            }
        });
    </script>
</body>
